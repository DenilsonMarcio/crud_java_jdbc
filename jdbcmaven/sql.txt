CREATE DATABASE digital_innovation_one; 

CREATE TABLE curso( 
   id INTEGER PRIMARY KEY AUTO_INCREMENT, 
   nome VARCHAR(100) NOT NULL, 
   duracao_horas DOUBLE NOT NULL 
); 

CREATE TABLE aluno( 
   id INTEGER PRIMARY KEY AUTO_INCREMENT, 
   nome VARCHAR(100) NOT NULL, 
   idade INT NOT NULL 
   estado CHAR(2) NOT NULL 
);
